Composite Dedup Logic

graph TD
    All["All ValidatedListings\nfrom all sources"] --> Check{"Same property?"}

    Check -->|"Address match\nOR geo ≤50m + price ±10%"| Cluster["Cluster together"]
    Check -->|"No match"| Standalone["Keep as standalone"]

    Cluster --> Merge["Merge best fields\nby source priority"]

    subgraph Priority["Source Priority (high → low)"]
        direction LR
        S1["StreetEasy"] --> S2["Zillow"] --> S3["Realtor"] --> S4["Apartments"] --> S5["RentHop"] --> S6["Craigslist"] --> S7["Facebook"]
    end

    Merge --> Fields

    subgraph Fields["Field Selection Rules"]
        direction TB
        F1["address, area → highest priority non-empty"]
        F2["price → mode across sources"]
        F3["beds, baths → highest priority where quality ≠ missing"]
        F4["lat, lon → geo priority order, skip 0,0"]
        F5["photos → union all URLs, dedup, cap at 20"]
        F6["dates → earliest list_date, latest update"]
        F7["quality → best confidence per field"]
    end

    Standalone --> Output["Final ValidatedListing[]"]
    Fields --> Output
